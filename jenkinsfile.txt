pipeline {
    agent { label 'master'}
    triggers {
        cron('*/1 * * * *')
    }
    stages {
        stage('scm') {
            steps {
                git 'https://github.com/jithinchacko/mavenjavacode.git'
            }
        }
        stage('build') {
            steps{
                sh 'mvn package'
            }
        }
        stage('postbuild') {
            steps{
                archiveArtifacts "target/*.jar"
            }
        }
    }
}
